CREATE TABLE users
      (
      id           TEXT    PRIMARY KEY,
      email           TEXT    NOT NULL,
      name            TEXT    NOT NULL,
      phone_number            TEXT    NOT NULL,
      password            TEXT    NOT NULL,
      facebook            TEXT,
      instagram            TEXT,
      date            timestamp     NOT NULL,
      ip_address            TEXT    NOT NULL);
CREATE INDEX email_index
ON users(email);
CREATE INDEX name_index
ON users(name);
CREATE INDEX phone_index
ON users(phone_number);
CREATE INDEX password_index
ON users(password);
CREATE INDEX ip_index
ON users(ip_address);

CREATE INDEX facebook_index
ON users(facebook);
CREATE INDEX instagram_index
ON users(instagram);

CREATE TABLE user_posts
      (
            id           TEXT    PRIMARY KEY,
            email           TEXT     NOT NULL,
            name           TEXT     NOT NULL,
            title            TEXT    NOT NULL,
            phone            TEXT    NOT NULL,
            animal            TEXT    NOT NULL,
            breed            TEXT    NOT NULL,
            price            INT    NOT NULL,
            age            INT    NOT NULL,
            age_type            TEXT    NOT NULL,
            description            TEXT,
            post_type            TEXT    NOT NULL,
            img_path            TEXT []   NOT NULL
      );
CREATE INDEX email_index
ON user_posts(email);

CREATE INDEX title_index
ON user_posts(title);

CREATE INDEX animal_index
ON user_posts(animal);

CREATE INDEX breed_index
ON user_posts(breed);

CREATE INDEX price_index
ON user_posts(price);

CREATE INDEX age_index
ON user_posts(age);

CREATE INDEX ageType_index
ON user_posts(age_type);

CREATE INDEX postType_index
ON user_posts(post_type);

CREATE INDEX imgPath_index
ON user_posts(img_path);
